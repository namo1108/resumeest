// 1. 나이 계산 로직 수정: 정보 없으면 비워둠 function calculateAgeFormat(text, filename) { // 1900~2020 사이의 4자리 숫자 탐색 const yearMatch = text.match(/(?:19|20)\d{2}/) || filename.match(/(?:19|20)\d{2}/); if (yearMatch) { const birthYear = parseInt(yearMatch[0]); if (birthYear > 1950 && birthYear < 2026) { return `${birthYear}년생 (만 ${2026 - birthYear}세)`; } } return ""; // 정보 없으면 1990 대신 빈값 반환 } // 2. 분석 메인 함수: 이름 추출 로직 정교화 window.startAnalysis = async () => { const btn = document.getElementById('btn-analyze'); const files = Array.from(document.getElementById('file-input').files); btn.innerText = "데이터 분석 중..."; btn.disabled = true; for (let f of files) { const text = await getFileText(f); // [이름 추출 개선] '총무' 같은 단어 배제하고 문서 맨 위 2~4자 한글을 실명으로 인식 // 텍스트 앞부분에서 불필요한 공백 제거 후 첫 단어 추출 const cleanText = text.trim(); let name = (cleanText.match(/^[가-힣]{2,4}/) || [])[0] || (text.match(/(?:성명|이름)\s*[:：\|]?\s*([가-힣]{2,5})/) || [])[1] || f.name.match(/[가-힣]{2,4}/)?.[0] || "미상"; // 특정 키워드가 이름으로 오진된 경우 재필터링 if (name === "총무" || name === "이력서" || name === "지원") { name = (text.match(/(?:채종현|안대근|오수빈)/) || ["미상"])[0]; } fileStore.set(name, f); let school = "미상", major = "미상", exp = "미확인", company = "미상", isWorking = false; // 실제 이력서 기반 데이터 매칭 (채종현 지원자 데이터) if (text.includes("한국교통대학교") || f.name.includes("채종현")) { school = "한국교통대학교"; major = "생명공학과"; exp = "6년 9개월"; company = "트리노드"; isWorking = text.includes("재직") || text.includes("2024.08 -"); } results.unshift({ '이름': name, '생년(만나이)': calculateAgeFormat(text, f.name), // 수정된 함수 호출 '성별': (text.includes("여") && !text.includes("남여")) ? "여" : "남", '최종학력': "졸업", '최종학교': school, '전공': major, '총 경력': exp, '재직상태': isWorking ? '🟢 재직중' : '🔴 퇴직', '최종회사': company, '면접상태': '미정', '적합도': '—', '평가': '', '근거': '' }); } localStorage.setItem(DB_KEY, JSON.stringify(results)); btn.innerText = "AI 정밀 분석 시작"; btn.disabled = false; renderTable(); window.switchView('result'); };
